    @Test
    void testCreateOrganizationProject_Success() {
        // Given
        String organizationId = "org123";
        String projectName = "New Project";
        String projectCode = "NEW001";
        String projectDescription = "New Description";
        
        when(securityContextUtil.getCurrentUsername()).thenReturn("testuser");
        when(organizationSecurityService.hasOrganizationAdminRole(organizationId, "testuser")).thenReturn(true);
        when(userRepository.findByUsername("testuser")).thenReturn(Optional.of(testUser));
        when(organizationRepository.findById(organizationId)).thenReturn(Optional.of(testOrganization));
        when(projectRepository.save(any(Project.class))).thenReturn(testProject);
        when(projectUserRepository.save(any(ProjectUser.class))).thenReturn(testProjectUser);

        // When
        Project result = projectService.createOrganizationProject(organizationId, projectName, projectCode, projectDescription);

        // Then
        assertNotNull(result);
        assertEquals(testProject.getId(), result.getId());
        
        // 조직 권한 확인
        verify(organizationSecurityService).hasOrganizationAdminRole(organizationId, "testuser");
        
        // 프로젝트 저장 확인
        verify(projectRepository).save(argThat(project -> 
            projectName.equals(project.getName()) && 
            projectCode.equals(project.getCode()) &&
            projectDescription.equals(project.getDescription()) &&
            project.getOrganization().equals(testOrganization)
        ));
        
        // 생성자가 PROJECT_MANAGER로 추가되는지 확인
        verify(projectUserRepository).save(argThat(projUser -> 
            projUser.getUser().equals(testUser) && 
            projUser.getRoleInProject() == ProjectUser.ProjectRole.PROJECT_MANAGER
        ));
        
        // 감사 로그 기록 확인
        verify(auditService, times(2)).logProjectAction(any(), any(), any());
        verify(auditService).logProjectMemberAction(any(), any(), any(), any());
    }

    // ==================== 프로젝트 이전 테스트 ====================

    @Test
    void testTransferProject_Success() {
        // Given
        String projectId = "proj123";
        String newOrganizationId = "neworg123";
        
        Organization newOrganization = new Organization();
        newOrganization.setId(newOrganizationId);
        newOrganization.setName("New Organization");
        
        when(securityContextUtil.getCurrentUsername()).thenReturn("testuser");
        when(projectSecurityService.hasProjectManagerRole(projectId, "testuser")).thenReturn(true);
        when(organizationSecurityService.hasOrganizationAdminRole(newOrganizationId, "testuser")).thenReturn(true);
        when(projectRepository.findById(projectId)).thenReturn(Optional.of(testProject));
        when(organizationRepository.findById(newOrganizationId)).thenReturn(Optional.of(newOrganization));
        when(projectRepository.save(any(Project.class))).thenReturn(testProject);

        // When
        Project result = projectService.transferProject(projectId, newOrganizationId);

        // Then
        assertNotNull(result);
        
        verify(projectSecurityService).hasProjectManagerRole(projectId, "testuser");
        verify(organizationSecurityService).hasOrganizationAdminRole(newOrganizationId, "testuser");
        verify(projectRepository).save(argThat(project -> 
            project.getOrganization().equals(newOrganization)
        ));
        verify(auditService).logProjectAction(eq(projectId), any(), contains("Project transferred"));
    }

    // ==================== 멤버 초대 테스트 ====================

    @Test
    void testInviteMember_Success() {
        // Given
        String projectId = "proj123";
        String username = "newuser";
        ProjectUser.ProjectRole role = ProjectUser.ProjectRole.DEVELOPER;
        
        User invitedUser = new User();
        invitedUser.setId("newuser123");
        invitedUser.setUsername(username);
        
        when(securityContextUtil.getCurrentUsername()).thenReturn("testuser");
        when(projectSecurityService.canInviteMembers(projectId, "testuser")).thenReturn(true);
        when(userRepository.findByUsername(username)).thenReturn(Optional.of(invitedUser));
        when(projectRepository.findById(projectId)).thenReturn(Optional.of(testProject));
        when(projectUserRepository.findByProjectIdAndUserId(projectId, "newuser123"))
            .thenReturn(Optional.empty());
        when(projectUserRepository.save(any(ProjectUser.class))).thenReturn(testProjectUser);

        // When
        ProjectUser result = projectService.inviteMember(projectId, username, role);

        // Then
        assertNotNull(result);
        
        verify(projectSecurityService).canInviteMembers(projectId, "testuser");
        verify(projectUserRepository).save(argThat(projUser -> 
            projUser.getUser().equals(invitedUser) && projUser.getRoleInProject() == role
        ));
        verify(auditService).logProjectMemberAction(eq(projectId), eq("newuser123"), any(), eq(role.toString()));
    }

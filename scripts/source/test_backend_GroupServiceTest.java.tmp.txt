    // ==================== 조직 그룹 생성 테스트 ====================

    @Test
    void testCreateOrganizationGroup_Success() {
        // Given
        String groupName = "New Organization Group";
        String groupDescription = "Test Description";
        String organizationId = "org123";
        
        when(securityContextUtil.getCurrentUsername()).thenReturn("testuser");
        when(organizationSecurityService.hasOrganizationAdminRole(organizationId, "testuser")).thenReturn(true);
        when(userRepository.findByUsername("testuser")).thenReturn(Optional.of(testUser));
        when(organizationRepository.findById(organizationId)).thenReturn(Optional.of(testOrganization));
        when(groupRepository.save(any(Group.class))).thenReturn(testOrganizationGroup);
        when(groupMemberRepository.save(any(GroupMember.class))).thenReturn(testGroupMember);

        // When
        Group result = groupService.createOrganizationGroup(organizationId, groupName, groupDescription);

        // Then
        assertNotNull(result);
        assertEquals(testOrganizationGroup.getId(), result.getId());
        
        // 조직 권한 확인
        verify(organizationSecurityService).hasOrganizationAdminRole(organizationId, "testuser");
        
        // 그룹 저장 확인
        verify(groupRepository).save(argThat(group -> 
            groupName.equals(group.getName()) && 
            groupDescription.equals(group.getDescription()) &&
            group.getOrganization().equals(testOrganization) &&
            group.getProject() == null
        ));
        
        // 생성자가 LEADER로 추가되는지 확인
        verify(groupMemberRepository).save(argThat(member -> 
            member.getUser().equals(testUser) && 
            member.getRoleInGroup() == GroupMember.GroupRole.LEADER
        ));
        
        // 감사 로그 기록 확인
        verify(auditService).logGroupAction(any(), any(), any());
        verify(auditService).logGroupMemberAction(any(), any(), any(), any());
    }

    // ==================== 그룹 멤버 초대 테스트 ====================

    @Test
    void testInviteMember_Success() {
        // Given
        String groupId = "group123";
        String username = "newuser";
        GroupMember.GroupRole role = GroupMember.GroupRole.MEMBER;
        
        User invitedUser = new User();
        invitedUser.setId("newuser123");
        invitedUser.setUsername(username);
        
        when(securityContextUtil.getCurrentUsername()).thenReturn("testuser");
        when(groupSecurityService.canInviteMembers(groupId, "testuser")).thenReturn(true);
        when(userRepository.findByUsername(username)).thenReturn(Optional.of(invitedUser));
        when(groupRepository.findById(groupId)).thenReturn(Optional.of(testOrganizationGroup));
        when(groupMemberRepository.findByGroupIdAndUserId(groupId, "newuser123"))
            .thenReturn(Optional.empty());
        when(groupMemberRepository.save(any(GroupMember.class))).thenReturn(testGroupMember);

        // When
        GroupMember result = groupService.inviteMember(groupId, username, role);

        // Then
        assertNotNull(result);
        
        verify(groupSecurityService).canInviteMembers(groupId, "testuser");
        verify(groupMemberRepository).save(argThat(member -> 
            member.getUser().equals(invitedUser) && member.getRoleInGroup() == role
        ));
        verify(auditService).logGroupMemberAction(eq(groupId), eq("newuser123"), any(), eq(role.toString()));
    }

    // ==================== 그룹 멤버 역할 변경 테스트 ====================

    @Test
    void testUpdateMemberRole_Success() {
        // Given
        String groupId = "group123";
        String targetUserId = "target123";
        GroupMember.GroupRole newRole = GroupMember.GroupRole.CO_LEADER;
        GroupMember.GroupRole oldRole = GroupMember.GroupRole.MEMBER;
        
        testGroupMember.setRoleInGroup(oldRole);
        
        when(securityContextUtil.getCurrentUsername()).thenReturn("testuser");
        when(groupSecurityService.hasLeadershipRole(groupId, "testuser")).thenReturn(true);
        when(groupMemberRepository.findByGroupIdAndUserId(groupId, targetUserId))
            .thenReturn(Optional.of(testGroupMember));
        when(groupMemberRepository.save(any(GroupMember.class))).thenReturn(testGroupMember);

        // When
        GroupMember result = groupService.updateMemberRole(groupId, targetUserId, newRole);

        // Then
        assertNotNull(result);
        
        verify(groupSecurityService).hasLeadershipRole(groupId, "testuser");
        verify(groupMemberRepository).save(argThat(member -> 
            member.getRoleInGroup() == newRole
        ));
        verify(auditService).logGroupMemberAction(eq(groupId), eq(targetUserId), any(), 
            contains("oldRole=" + oldRole + ", newRole=" + newRole));
    }
